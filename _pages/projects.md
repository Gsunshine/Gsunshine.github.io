<script>
  (function () {
    var params = new URLSearchParams(window.location.search);
    var venue = (params.get("venue") || "").toLowerCase();
    var year = (params.get("year") || "").toLowerCase();
    if (!venue && !year) return;

    var content = document.querySelector(".page__content");
    if (!content) return;
    content.classList.add("projects-is-filtered");

    var anyVisible = false;
    var lists = Array.prototype.slice.call(content.querySelectorAll("ul"));
    var venueLabelMap = {
      arxiv: "arXiv",
      cvpr: "CVPR",
      neurips: "NeurIPS",
      iclr: "ICLR",
      icml: "ICML",
      eccv: "ECCV",
      tip: "TIP",
      report: "Tech Report"
    };

    lists.forEach(function (ul) {
      var items = Array.prototype.slice.call(ul.querySelectorAll("li"));
      var visibleInList = 0;

      items.forEach(function (li) {
        var meta = li.querySelector(".proj-filter");
        if (!meta) return;

        var itemVenue = (meta.getAttribute("data-venue") || "").toLowerCase();
        var itemYear = (meta.getAttribute("data-year") || "").toLowerCase();
        var matched = (!venue || itemVenue === venue) && (!year || itemYear === year);

        li.style.display = matched ? "" : "none";
        if (matched) {
          visibleInList += 1;
          anyVisible = true;
        }
      });

      ul.style.display = visibleInList > 0 ? "" : "none";
    });

    var headings = Array.prototype.slice.call(content.querySelectorAll("h3"));
    headings.forEach(function (h3) {
      var node = h3.nextElementSibling;
      var hasVisible = false;
      while (node && node.tagName !== "H3") {
        if (node.tagName === "UL" && node.style.display !== "none") {
          hasVisible = true;
          break;
        }
        node = node.nextElementSibling;
      }
      h3.style.display = hasVisible ? "" : "none";
    });

    var note = document.getElementById("projects-filter-note");
    if (!note) return;
    var clearHref = "{{ site.baseurl }}/#selected-works";

    if (!anyVisible) {
      note.style.display = "";
      note.innerHTML = 'No matches for current filter. <a href="' + clearHref + '">Clear</a>';
      return;
    }

    var parts = [];
    if (venue) parts.push("venue: " + (venueLabelMap[venue] || venue));
    if (year) parts.push("year: " + year);
    note.style.display = "";
    note.innerHTML = 'Filtered by <strong>' + parts.join(", ") + '</strong>. <a href="' + clearHref + '">Clear</a>';
  })();
</script>
